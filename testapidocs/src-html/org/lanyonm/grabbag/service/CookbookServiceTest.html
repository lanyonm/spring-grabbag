001    package org.lanyonm.grabbag.service;
002    
003    import static org.junit.Assert.assertEquals;
004    import static org.junit.Assert.assertFalse;
005    import static org.junit.Assert.assertNotNull;
006    import static org.junit.Assert.assertNull;
007    import static org.junit.Assert.assertTrue;
008    
009    import java.util.ArrayList;
010    import java.util.List;
011    
012    import org.junit.Test;
013    import org.junit.runner.RunWith;
014    import org.lanyonm.grabbag.config.DataConfig;
015    import org.lanyonm.grabbag.domain.Ingredient;
016    import org.lanyonm.grabbag.domain.Recipe;
017    import org.lanyonm.grabbag.domain.RecipeIngredient;
018    import org.springframework.beans.factory.annotation.Autowired;
019    import org.springframework.test.context.ContextConfiguration;
020    import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
021    import org.springframework.test.context.support.AnnotationConfigContextLoader;
022    
023    @RunWith(SpringJUnit4ClassRunner.class)
024    @ContextConfiguration(loader=AnnotationConfigContextLoader.class, classes={DataConfig.class, CookbookServiceImpl.class})
025    public class CookbookServiceTest {
026    
027            @Autowired
028            private CookbookService cookbookService;
029    
030            @Test
031            public void testGetAllRecipes() {
032                    List<Recipe> recipes = cookbookService.getAllRecipes();
033                    assertNotNull("recipes null", recipes);
034                    assertEquals("number of recipes", 2, recipes.size());
035            }
036    
037            @Test
038            public void testDeleteIngredient() {
039                    Ingredient ingredient = cookbookService.getIngredient(2);
040                    assertEquals("should have butter", "butter", ingredient.getName());
041                    assertTrue(cookbookService.deleteIngredient(ingredient));
042                    assertNull("should have null", cookbookService.getIngredient(2));
043            }
044    
045            @Test
046            public void testDeleteIngredientWithDependency() {
047                    Ingredient ingredient = cookbookService.getIngredient(8);
048                    assertEquals("should have Cool Whip", "Cool Whip", ingredient.getName());
049                    assertFalse(cookbookService.deleteIngredient(ingredient));
050            }
051    
052            @Test
053            public void testInsertRecipeSuccess() {
054                    Recipe recipe = new Recipe();
055                    recipe.setName("instant cream cheese pudding");
056                    List<Ingredient> ingredients = new ArrayList<Ingredient>();
057                    RecipeIngredient recipeIngredient = new RecipeIngredient();
058                    recipeIngredient.setId(7);
059                    recipeIngredient.setVolume("1/4 cup");
060                    ingredients.add(recipeIngredient);
061                    recipeIngredient = new RecipeIngredient();
062                    recipeIngredient.setId(9);
063                    recipeIngredient.setVolume("2 pkg");
064                    ingredients.add(recipeIngredient);
065                    recipe.setIngredients(ingredients);
066                    ingredients.add(recipeIngredient);
067                    assertTrue("only 1 row should have been affected", cookbookService.saveRecipe(recipe));
068                    recipe = cookbookService.getRecipe(3);
069                    assertNotNull(recipe);
070                    assertEquals("recipe name should be 'instant cream cheese pudding'", "instant cream cheese pudding", cookbookService.getRecipe(3).getName());
071            }
072    
073            @Test
074            public void testDeleteRecipeSuccess() {
075                    Recipe recipe = cookbookService.getRecipe(3);
076                    assertNotNull(recipe);
077                    assertEquals("should be 3 recipes", 3, cookbookService.getAllRecipes().size());
078                    assertTrue("should have deleted recipe", cookbookService.deleteRecipe(recipe));
079                    assertEquals("should be 2 recipes", 2, cookbookService.getAllRecipes().size());
080            }
081    }
