023    /**
024     * Tests {@link RecipeMapper}

026     * @author lanyonm
027     */
028    @RunWith(SpringJUnit4ClassRunner.class)
029    @ContextConfiguration(loader=AnnotationConfigContextLoader.class, classes={DataConfig.class})
030    public class RecipeMapperTest {
031    
032            @Autowired
033            private RecipeMapper recipeMapper;
034    
035            /**
036             * @see RecipeMapper#getAllRecipes()
037             */
038            @Test
039            public void testGetAllRecipes() {
040                    List<Recipe> recipes = recipeMapper.getAllRecipes();
041                    assertNotNull(recipes);
042                    assertEquals("number of recipes", 2, recipes.size());
043            }
044    
045            /**
046             * Lookup Pistachio Dessert
047             * 
048             * @see RecipeMapper#getRecipe(int)
049             */
050            @Test
051            public void testGetRecipe() {
052                    Recipe recipe = recipeMapper.getRecipe(1);
053                    assertNotNull(recipe);
054                    assertEquals("recipe name", "Pistachio Dessert", recipe.getName());
055                    assertEquals("recipe description", null, recipe.getDescription());
056                    assertNotNull("recipe ingredients", recipe.getIngredients());
057                    assertEquals("number of ingredients", 9, recipe.getIngredients().size());
058                    for (Object ingredient : recipe.getIngredients()) {
059                            isInstanceOf(RecipeIngredient.class, ingredient, "ingredients are RecipeIngredients");
060                            assertNotNull(((RecipeIngredient) ingredient).getName());
061                            assertNotNull(((RecipeIngredient) ingredient).getName() + " volume", ((RecipeIngredient) ingredient).getVolume());
062                    }
063            }
064    
065            @Test
066            @Ignore("implement this")
067            public void testUpdateRecipe() {
068                    fail("test something");
069            }
070    
071            @Test
072            public void testInsertRecipeSuccess() {
073                    Recipe recipe = new Recipe();
074                    recipe.setName("instant cream cheese pudding");
075                    assertEquals("only 1 row should have been affected", 1, recipeMapper.insertRecipe(recipe));
076                    assertTrue("new recipe id should be greater than 2", recipe.getId() > 2);
077            }
078    
079            @Test
080            public void testInsertRecipeIngredientSuccess() {
081                    assertEquals("only 1 ingredient in recipeId 2", 1, recipeMapper.getRecipe(2).getIngredients().size());
082                    HashMap<String, Object> params = new HashMap<String, Object>();
083                    params.put("recipeId", 2);
084                    params.put("ingredientId", 2);
085                    params.put("volume", "2 sticks");
086                    assertEquals("only 1 row should have been affected", 1, recipeMapper.insertRecipeIngredient(params));
087                    assertEquals("should be 2 ingredients in recipeId 2", 2, recipeMapper.getRecipe(2).getIngredients().size());
088            }
089    
090            @Test
091            public void testDeleteRecipeSuccess() {
092                    assertEquals("there should be 3 recipes", 3, recipeMapper.getAllRecipes().size());
093                    Recipe recipe = recipeMapper.getRecipe((int) recipeMapper.getAllRecipes().get(2).getId());
094                    assertNotNull("recipe should not be null", recipe);
095                    assertEquals("only 1 row should be affected", 1, recipeMapper.deleteRecipe(recipe));
096                    assertEquals("there should be 2 recipes", 2, recipeMapper.getAllRecipes().size());
097            }
098    }
