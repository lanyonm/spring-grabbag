001    package org.lanyonm.grabbag.web.controller;
002    
003    import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.*;
004    import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.*;
005    
006    import org.junit.Before;
007    import org.junit.Test;
008    import org.junit.runner.RunWith;
009    
010    import org.springframework.beans.factory.annotation.Autowired;
011    import org.springframework.test.context.ContextConfiguration;
012    import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
013    import org.springframework.test.context.web.AnnotationConfigWebContextLoader;
014    import org.springframework.test.context.web.WebAppConfiguration;
015    import org.springframework.test.web.servlet.MockMvc;
016    import org.springframework.test.web.servlet.setup.MockMvcBuilders;
017    import org.springframework.web.context.WebApplicationContext;
018    
019    import org.lanyonm.grabbag.config.DataConfig;
020    import org.lanyonm.grabbag.config.ViewResolver;
021    import org.lanyonm.grabbag.config.WebConfig;
022    import org.lanyonm.grabbag.dao.jdbc.UsersDaoImpl;
023    
024    @RunWith(SpringJUnit4ClassRunner.class)
025    @WebAppConfiguration
026    @ContextConfiguration(loader=AnnotationConfigWebContextLoader.class, classes={DataConfig.class, WebConfig.class, ViewResolver.class, HomeController.class, UsersDaoImpl.class})
027    public class HomeControllerTest {
028    
029            @Autowired
030            private WebApplicationContext wac;
031    
032            private MockMvc mockMvc;
033    
034            @Before
035            public void setup() {
036                    this.mockMvc = MockMvcBuilders.webAppContextSetup(this.wac).build();
037            }
038    
039            @Test
040            public void testIndex() throws Exception {
041                    this.mockMvc.perform(get("/")).andExpect(status().isOk());
042            }
043    
044            @Test
045            public void testDefinedPath() throws Exception {
046                    this.mockMvc.perform(get("/defined-path")).andExpect(status().isOk())
047                            .andExpect(model().attribute("sample", "Defined Path!!!"));
048            }
049    
050            @Test
051            public void testJdbc() throws Exception {
052                    this.mockMvc.perform(get("/jdbc")).andExpect(status().isOk())
053                            .andExpect(model().size(1))
054                            .andExpect(model().attributeExists("users"));
055            }
056    
057            @Test
058            public void testMyBatis() throws Exception {
059                    this.mockMvc.perform(get("/mybatis")).andExpect(status().isOk())
060                            .andExpect(model().attributeExists("ingredients"));
061            }
062    }
