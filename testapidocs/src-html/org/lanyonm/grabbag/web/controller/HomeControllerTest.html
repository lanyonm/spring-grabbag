001    package org.lanyonm.grabbag.web.controller;
002    
003    import static org.junit.Assert.assertEquals;
004    import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.*;
005    import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.*;
006    
007    import java.util.List;
008    
009    import org.junit.Before;
010    import org.junit.Test;
011    import org.junit.runner.RunWith;
012    
013    import org.springframework.beans.factory.annotation.Autowired;
014    import org.springframework.test.context.ContextConfiguration;
015    import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
016    import org.springframework.test.context.web.AnnotationConfigWebContextLoader;
017    import org.springframework.test.context.web.WebAppConfiguration;
018    import org.springframework.test.web.servlet.MockMvc;
019    import org.springframework.test.web.servlet.MvcResult;
020    import org.springframework.test.web.servlet.setup.MockMvcBuilders;
021    import org.springframework.web.context.WebApplicationContext;
022    
023    import org.lanyonm.grabbag.config.DataConfig;
024    import org.lanyonm.grabbag.config.ViewResolver;
025    import org.lanyonm.grabbag.config.WebConfig;
026    import org.lanyonm.grabbag.dao.jdbc.UsersDaoImpl;
027    import org.lanyonm.grabbag.domain.User;
028    
029    @RunWith(SpringJUnit4ClassRunner.class)
030    @WebAppConfiguration
031    @ContextConfiguration(loader=AnnotationConfigWebContextLoader.class, classes={DataConfig.class, WebConfig.class, ViewResolver.class, HomeController.class, UsersDaoImpl.class})
032    public class HomeControllerTest {
033    
034            @Autowired
035            private WebApplicationContext wac;
036    
037            private MockMvc mockMvc;
038    
039            @Before
040            public void setup() {
041                    this.mockMvc = MockMvcBuilders.webAppContextSetup(this.wac).build();
042            }
043    
044            @Test
045            public void testIndex() throws Exception {
046                    this.mockMvc.perform(get("/")).andExpect(status().isOk());
047            }
048    
049            @Test
050            public void testDefinedPath() throws Exception {
051                    this.mockMvc.perform(get("/defined-path")).andExpect(status().isOk())
052                            .andExpect(model().attribute("sample", "Defined Path!!!"));
053            }
054    
055            @Test
056            public void testJdbc() throws Exception {
057                    MvcResult result = this.mockMvc.perform(get("/jdbc")).andExpect(status().isOk())
058                            .andExpect(model().size(1))
059                            .andExpect(model().attributeExists("users"))
060                            .andReturn();
061                    @SuppressWarnings("unchecked")
062                    List<User> users = (List<User>) result.getModelAndView().getModel().get("users");
063                    assertEquals("there should be two users", 2, users.size());
064                    assertEquals("first user's name should be 'Mike L. <lanyonm@gmail.com>'", "Mike L. <lanyonm@gmail.com>", users.get(0).toString());
065            }
066    
067            @Test
068            public void testMyBatis() throws Exception {
069                    this.mockMvc.perform(get("/mybatis")).andExpect(status().isOk())
070                            .andExpect(model().attributeExists("ingredients"));
071            }
072    }
