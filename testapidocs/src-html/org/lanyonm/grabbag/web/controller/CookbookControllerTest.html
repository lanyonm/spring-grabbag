105            @Test
106            public void testRecipeAddDelete() throws Exception {
107                    this.mockMvc.perform(post("/cookbook/recipe/{id}/edit", 0).param("name", "Test Recipe").param("description", "A recipe used during testing"))
108                                    .andExpect(redirectedUrl("/cookbook/"));
109                    MvcResult result = this.mockMvc.perform(get("/cookbook/"))
110                                    .andExpect(status().isOk())
111                                    .andExpect(model().attributeExists("recipes"))
112                                    .andReturn();
113                    @SuppressWarnings("unchecked")
114                    List<Recipe> recipes = (List<Recipe>) result.getModelAndView().getModel().get("recipes");
115                    assertEquals("there should be three recipes", recipes.size(), 3);
116                    long recipeId = recipes.get(recipes.size() -1).getId();
117                    this.mockMvc.perform(get("/cookbook/recipe/{id}/delete", recipeId))
118                                    .andExpect(redirectedUrl("/cookbook/?message=You+successfully+deleted+Test+Recipe%21"));
119                    result = this.mockMvc.perform(get("/cookbook/?message=You+successfully+deleted+Test+Recipe%21"))
120                                    .andExpect(status().isOk())
121                                    .andExpect(model().attributeExists("recipes"))
122                                    .andReturn();
123                    @SuppressWarnings("unchecked")
124                    List<Recipe> recipes2 = (List<Recipe>) result.getModelAndView().getModel().get("recipes");
125                    assertEquals("there should be two recipes", recipes2.size(), 2);
126            }

175            @Test
176            public void testIngredientAddDelete() throws Exception {
177                    this.mockMvc.perform(post("/cookbook/ingredient/{id}/edit", 0).param("name", "test ingredient").param("description", "this is a test ingredient"))
178                                    .andExpect(redirectedUrl("/cookbook/ingredients"));
179                    MvcResult result = this.mockMvc.perform(get("/cookbook/ingredients"))
180                                    .andExpect(status().isOk())
181                                    .andExpect(model().attributeExists("ingredients"))
182                                    .andReturn();
183                    @SuppressWarnings("unchecked")
184                    List<Ingredient> ingredients = (List<Ingredient>) result.getModelAndView().getModel().get("ingredients");
185                    long ingredientId = ingredients.get(ingredients.size() - 1).getId();
186                    assertTrue("ingredientId should be be a positive int", ingredientId > 0);
187                    this.mockMvc.perform(get("/cookbook/ingredient/{id}/delete", ingredientId))
188                                    .andExpect(redirectedUrl("/cookbook/ingredients?message=You+successfully+deleted+test+ingredient%21"));
189                    this.mockMvc.perform(post("/cookbook/ingredient/{id}/edit", 0).param("name", "").param("description", "this is a test ingredient"))
190                                    .andExpect(status().isOk())
191                                    .andExpect(model().size(2))
192                                    .andExpect(model().attributeErrorCount("ingredient", 1))
193                                    .andExpect(model().attributeHasFieldErrors("ingredient", "name"))
194                                    .andExpect(model().attribute("isNew", true))
195                                    .andExpect(view().name("cookbook/ingredientAddEdit"));
196            }

209            @Test
210            public void testGraphJSON() throws Exception {
211                    MvcResult result = this.mockMvc.perform(get("/cookbook/graph.json"))
212                                    .andExpect(status().isOk())
213                                    .andExpect(model().attributeExists("nodes", "links"))
214                                    .andExpect(content().contentType(MediaType.APPLICATION_JSON))
215                                    .andReturn();
216                    List<Node> nodes = (List<Node>) result.getModelAndView().getModel().get("nodes");
217                    assertEquals("first node name should be first recipe", "Pistachio Dessert", nodes.get(0).getName());
218                    assertEquals("first node's group should be one", 1, nodes.get(0).getGroup());
219                    List<Link> links = (List<Link>) result.getModelAndView().getModel().get("links");
220                    assertEquals("first link's source should be 0", 0, links.get(0).getSource());
221                    assertEquals("first link's target should be 7", 7, links.get(0).getTarget());
222                    assertEquals("first link's value should be 1", 1, links.get(0).getValue());
