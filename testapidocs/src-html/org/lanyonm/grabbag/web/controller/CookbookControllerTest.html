001    package org.lanyonm.grabbag.web.controller;
002    
003    import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
004    import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.post;
005    import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;
006    import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.model;
007    import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.redirectedUrl;
008    import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
009    import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.view;
010    import static org.springframework.test.web.servlet.result.MockMvcResultHandlers.print;
011    import static org.springframework.test.web.servlet.setup.MockMvcBuilders.webAppContextSetup;
012    import static org.hamcrest.CoreMatchers.containsString;
013    import static org.junit.Assert.assertTrue;
014    
015    import java.util.List;
016    import java.util.Locale;
017    
018    import org.junit.Before;
019    import org.junit.Test;
020    import org.junit.runner.RunWith;
021    import org.lanyonm.grabbag.config.DataConfig;
022    import org.lanyonm.grabbag.config.ViewResolver;
023    import org.lanyonm.grabbag.config.WebConfig;
024    import org.lanyonm.grabbag.domain.Ingredient;
025    import org.lanyonm.grabbag.service.CookbookServiceImpl;
026    import org.springframework.beans.factory.annotation.Autowired;
027    import org.springframework.http.MediaType;
028    import org.springframework.test.context.ContextConfiguration;
029    import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
030    import org.springframework.test.context.web.AnnotationConfigWebContextLoader;
031    import org.springframework.test.context.web.WebAppConfiguration;
032    import org.springframework.test.web.servlet.MockMvc;
033    import org.springframework.test.web.servlet.MvcResult;
034    import org.springframework.web.context.WebApplicationContext;
035    
036    @RunWith(SpringJUnit4ClassRunner.class)
037    @WebAppConfiguration
038    @ContextConfiguration(loader=AnnotationConfigWebContextLoader.class, classes={DataConfig.class, WebConfig.class, ViewResolver.class,
039            CookbookController.class, CookbookServiceImpl.class})
040    public class CookbookControllerTest {
041    
042            @Autowired
043            private WebApplicationContext wac;
044    
045            private MockMvc mockMvc;
046    
047            @Before
048            public void setup() {
049                    this.mockMvc = webAppContextSetup(this.wac).build();
050            }
051    
052            @Test
053            public void testIndex() throws Exception {
054                    this.mockMvc.perform(get("/cookbook/"))
055                                    .andExpect(status().isOk())
056                                    .andExpect(view().name("cookbook/index"))
057                                    .andExpect(model().attributeExists("recipes"));
058            }
059    
060            @Test
061            public void testGetRecipe() throws Exception {
062                    this.mockMvc.perform(get("/cookbook/recipe/{id}", 1))
063                                    .andExpect(status().isOk())
064                                    .andExpect(view().name("cookbook/recipe"))
065                                    .andExpect(model().attributeExists("recipe"));
066                    this.mockMvc.perform(get("/cookbook/recipe/{id}", Integer.MAX_VALUE))
067                                    .andExpect(redirectedUrl("/cookbook/?error=Something+nefarious+was+attempted."));
068            }
069    
070            @Test
071            public void testRecipeAddEdit() throws Exception {
072                    this.mockMvc.perform(get("/cookbook/recipe/{id}/edit", Integer.MAX_VALUE))
073                                    .andExpect(redirectedUrl("/cookbook/?error=Something+nefarious+was+attempted."));
074                    this.mockMvc.perform(get("/cookbook/recipe/{id}/edit", 1))
075                                    .andExpect(status().isOk())
076                                    .andExpect(view().name("cookbook/recipeAddEdit"))
077                                    .andExpect(model().size(2))
078                                    .andExpect(model().attributeExists("recipe", "allIngredients"));
079                    this.mockMvc.perform(get("/cookbook/recipe/{id}/edit", 0))
080                                    .andExpect(status().isOk())
081                                    .andExpect(view().name("cookbook/recipeAddEdit"))
082                                    .andExpect(model().size(2))
083                                    .andExpect(model().attributeExists("recipe", "allIngredients"));
084            }
085    
086            @Test
087            public void testIngredients() throws Exception {
088                    this.mockMvc.perform(get("/cookbook/ingredients"))
089                                    .andExpect(status().isOk())
090                                    .andExpect(view().name("cookbook/ingredients"))
091                                    .andExpect(model().attributeExists("ingredients"));
092            }
093    
094            @Test
095            public void testIngredientEdit() throws Exception {
096                    this.mockMvc.perform(get("/cookbook/ingredient/{id}/edit", 1).locale(Locale.FRENCH).accept(MediaType.TEXT_HTML))
097                                    .andDo(print())
098                                    .andExpect(status().isOk())
099                                    .andExpect(view().name("cookbook/ingredientAddEdit"))
100                                    .andExpect(model().size(2))
101                                    .andExpect(model().attributeExists("ingredient", "isNew"));
102                                    // We cannot test the jsps and therefore don't have a way to test i18n
103                                    // .andExpect(content().string(containsString("Nom")));
104            }
105    
106            @Test
107            public void testIngredientEditPost() throws Exception {
108                    this.mockMvc.perform(post("/cookbook/ingredient/{id}/edit", 1).param("name", "milk").param("description", "it comes from cows!"))
109                                    .andExpect(redirectedUrl("/cookbook/ingredients"));
110                    this.mockMvc.perform(post("/cookbook/ingredient/{id}/edit", 1).param("name", "").param("description", "it comes from cows!"))
111                                    .andExpect(status().isOk())
112                                    .andExpect(model().attributeErrorCount("ingredient", 1))
113                                    .andExpect(model().attributeHasFieldErrors("ingredient", "name"));
114            }
115    
116            @Test
117            public void testIngredientAddDelete() throws Exception {
118                    this.mockMvc.perform(post("/cookbook/ingredient/{id}/edit", 0).param("name", "test ingredient").param("description", "this is a test ingredient"))
119                                    .andExpect(redirectedUrl("/cookbook/ingredients"));
120                    MvcResult result = this.mockMvc.perform(get("/cookbook/ingredients"))
121                                    .andExpect(status().isOk())
122                                    .andExpect(model().attributeExists("ingredients"))
123                                    .andReturn();
124                    @SuppressWarnings("unchecked")
125                    List<Ingredient> ingredients = (List<Ingredient>) result.getModelAndView().getModel().get("ingredients");
126                    long ingredientId = ingredients.get(ingredients.size() - 1).getId();
127                    assertTrue("ingredientId should be be a positive int", ingredientId > 0);
128                    this.mockMvc.perform(get("/cookbook/ingredient/{id}/delete", ingredientId))
129                                    .andExpect(redirectedUrl("/cookbook/ingredients?message=You+successfully+deleted+test+ingredient%21"));
130            }
131    
132    }
