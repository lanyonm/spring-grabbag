054            @Test

056                    this.mockMvc.perform(get("/cookbook/"))

063            public void testGetRecipe() throws Exception {
064                    this.mockMvc.perform(get("/cookbook/recipe/{id}", 1))
065                                    .andExpect(status().isOk())
066                                    .andExpect(view().name("cookbook/recipe"))
067                                    .andExpect(model().attributeExists("recipe"));

072            @Test
073            public void testRecipeAddEdit() throws Exception {
074                    this.mockMvc.perform(get("/cookbook/recipe/{id}/edit", Integer.MAX_VALUE))
075                                    .andExpect(redirectedUrl("/cookbook/?error=Something+nefarious+was+attempted."));
076                    this.mockMvc.perform(get("/cookbook/recipe/{id}/edit", 1))
077                                    .andExpect(status().isOk())
078                                    .andExpect(view().name("cookbook/recipeAddEdit"))
079                                    .andExpect(model().size(2))
080                                    .andExpect(model().attributeExists("recipe", "allIngredients"));
081                    this.mockMvc.perform(get("/cookbook/recipe/{id}/edit", 0))
082                                    .andExpect(status().isOk())
083                                    .andExpect(view().name("cookbook/recipeAddEdit"))
084                                    .andExpect(model().size(2))
085                                    .andExpect(model().attributeExists("recipe", "allIngredients"));
086            }

088            @Test
089            public void testRecipeEditPost() throws Exception {
090                    this.mockMvc.perform(post("/cookbook/recipe/{id}/edit", Integer.MAX_VALUE))
091                                    .andExpect(redirectedUrl("/cookbook/?error=Something+nefarious+was+attempted."));
092                    Recipe testRecipe = new Recipe();
093                    testRecipe.setId(1);
094                    testRecipe.setName("Pistachio Dessert");
095                    this.mockMvc.perform(get("/cookbook/recipe/{id}/edit", 1))
096                                    .andExpect(status().isOk())
097                                    .andExpect(model().attribute("recipe", testRecipe));
098                    this.mockMvc.perform(post("/cookbook/recipe/{id}/edit", 1).param("name", "Pistachio Dessert").param("description", "test description"))
099                                    .andExpect(redirectedUrl("/cookbook/"));
100    
101            }
102    
103            @Test
104            public void testRecipeAddDelete() throws Exception {
105                    this.mockMvc.perform(post("/cookbook/recipe/{id}/edit", 0).param("name", "Test Recipe").param("description", "A recipe used during testing"))
106                                    .andExpect(redirectedUrl("/cookbook/"));
107                    MvcResult result = this.mockMvc.perform(get("/cookbook/"))
108                                    .andExpect(status().isOk())
109                                    .andExpect(model().attributeExists("recipes"))
110                                    .andReturn();
111                    @SuppressWarnings("unchecked")
112                    List<Recipe> recipes = (List<Recipe>) result.getModelAndView().getModel().get("recipes");
113                    assertEquals("there should be three recipes", recipes.size(), 3);
114                    long recipeId = recipes.get(recipes.size() -1).getId();
115                    this.mockMvc.perform(get("/cookbook/recipe/{id}/delete", recipeId))
116                                    .andExpect(redirectedUrl("/cookbook/?message=You+successfully+deleted+Test+Recipe%21"));
117                    result = this.mockMvc.perform(get("/cookbook/?message=You+successfully+deleted+Test+Recipe%21"))
118                                    .andExpect(status().isOk())
119                                    .andExpect(model().attributeExists("recipes"))
120                                    .andReturn();
121                    @SuppressWarnings("unchecked")
122                    List<Recipe> recipes2 = (List<Recipe>) result.getModelAndView().getModel().get("recipes");
123                    assertEquals("there should be two recipes", recipes2.size(), 2);
124            }

152            @Test
153            public void testIngredientEditPost() throws Exception {
154                    this.mockMvc.perform(post("/cookbook/ingredient/{id}/edit", 1).param("name", "milk").param("description", "it comes from cows!"))
155                                    .andExpect(redirectedUrl("/cookbook/ingredients"));

162            @Test
163            public void testIngredientAddDelete() throws Exception {
164                    this.mockMvc.perform(post("/cookbook/ingredient/{id}/edit", 0).param("name", "test ingredient").param("description", "this is a test ingredient"))
165                                    .andExpect(redirectedUrl("/cookbook/ingredients"));
166                    MvcResult result = this.mockMvc.perform(get("/cookbook/ingredients"))
167                                    .andExpect(status().isOk())
168                                    .andExpect(model().attributeExists("ingredients"))
169                                    .andReturn();
170                    @SuppressWarnings("unchecked")
171                    List<Ingredient> ingredients = (List<Ingredient>) result.getModelAndView().getModel().get("ingredients");
172                    long ingredientId = ingredients.get(ingredients.size() - 1).getId();
173                    assertTrue("ingredientId should be be a positive int", ingredientId > 0);
174                    this.mockMvc.perform(get("/cookbook/ingredient/{id}/delete", ingredientId))
175                                    .andExpect(redirectedUrl("/cookbook/ingredients?message=You+successfully+deleted+test+ingredient%21"));
176            }
