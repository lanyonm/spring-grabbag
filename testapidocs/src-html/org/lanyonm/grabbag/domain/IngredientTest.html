001    package org.lanyonm.grabbag.domain;
002    
003    import static org.hamcrest.CoreMatchers.equalTo;
004    import static org.hamcrest.CoreMatchers.not;
005    import static org.hamcrest.MatcherAssert.assertThat;
006    
007    import org.junit.Before;
008    import org.junit.Test;
009    
010    public class IngredientTest {
011    
012            private Ingredient one;
013            private Ingredient two;
014    
015            @Before
016            public void setup() {
017                    one = new Ingredient();
018                    one.setName("Test");
019                    two = new Ingredient();
020                    two.setName("Test");
021            }
022    
023            @Test
024            public void testEquals() {
025                    assertThat(one, equalTo(one));
026                    assertThat("ingredients with the same name and description should be equal", one, equalTo(two));
027                    one.setDescription("desc");
028                    two.setDescription("desc");
029                    assertThat("ingredients with the same name and description should be equal", one, equalTo(two));
030            }
031    
032            @Test
033            public void testNotEquals() {
034                    assertThat("recipe compare to null", one, not(equalTo(null)));
035                    assertThat("recipe compare to non-recipe", one, not(equalTo(new Object())));
036                    Ingredient three = new Ingredient();
037                    assertThat("recipe compare to empty recipe", one, not(equalTo(three)));
038                    // id
039                    three.setId(10);
040                    assertThat("recipe compare to recipe with different id", one, not(equalTo(three)));
041                    three.setId(0);
042                    // name
043                    assertThat("recipe compare to recipe with different name", three, not(equalTo(one)));
044                    three.setName("three");
045                    assertThat("recipe compare to recipe with different name", one, not(equalTo(three)));
046                    three.setName("Test");
047                    // description
048                    one.setDescription("one");
049                    assertThat("recipe compare to recipe with different description", three, not(equalTo(one)));
050                    three.setDescription("desc");
051                    assertThat("recipe compare to recipe with different description", one, not(equalTo(three)));
052            }
053    }
