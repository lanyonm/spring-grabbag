001    package org.lanyonm.grabbag.domain;
002    
003    import static org.hamcrest.CoreMatchers.containsString;
004    import static org.hamcrest.CoreMatchers.equalTo;
005    import static org.hamcrest.CoreMatchers.not;
006    import static org.hamcrest.MatcherAssert.assertThat;
007    
008    import org.junit.Before;
009    import org.junit.Test;
010    
011    public class RecipeTest {
012    
013            private Recipe one;
014            private Recipe two;
015    
016            @Before
017            public void setup() {
018                    one = new Recipe();
019                    one.setName("Test");
020                    two = new Recipe();
021                    two.setName("Test");
022            }
023    
024            @Test
025            public void testEquals() {
026                    assertThat(one, equalTo(one));
027                    assertThat("recipes with the same name and description should be equal", one, equalTo(two));
028                    one.setDescription("desc");
029                    two.setDescription("desc");
030                    assertThat("recipes with the same name and description should be equal", one, equalTo(two));
031            }
032    
033            @Test
034            public void testNotEquals() {
035                    assertThat("recipe compare to null", one, not(equalTo(null)));
036                    assertThat("recipe compare to non-recipe", one, not(equalTo(new Object())));
037                    Recipe three = new Recipe();
038                    assertThat("recipe compare to empty recipe", one, not(equalTo(three)));
039                    // id
040                    three.setId(10);
041                    assertThat("recipe compare to recipe with different id", one, not(equalTo(three)));
042                    three.setId(0);
043                    // name
044                    assertThat("recipe compare to recipe with different name", three, not(equalTo(one)));
045                    three.setName("three");
046                    assertThat("recipe compare to recipe with different name", one, not(equalTo(three)));
047                    three.setName("Test");
048                    // description
049                    one.setDescription("one");
050                    assertThat("recipe compare to recipe with different description", three, not(equalTo(one)));
051                    three.setDescription("desc");
052                    assertThat("recipe compare to recipe with different description", one, not(equalTo(three)));
053            }
054    
055            @Test
056            public void testToString() {
057                    assertThat(one.toString(), containsString("Recipe [id=0, name=Test"));
058            }
059    }
