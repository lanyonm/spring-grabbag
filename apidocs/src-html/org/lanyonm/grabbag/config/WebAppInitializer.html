001    package org.lanyonm.grabbag.config;
002    
003    import java.util.Calendar;
004    import java.util.Set;
005    
006    import javax.servlet.FilterRegistration;
007    import javax.servlet.ServletContext;
008    import javax.servlet.ServletException;
009    import javax.servlet.ServletRegistration;
010    
011    import org.slf4j.Logger;
012    import org.slf4j.LoggerFactory;
013    import org.springframework.web.WebApplicationInitializer;
014    import org.springframework.web.context.ContextLoaderListener;
015    import org.springframework.web.context.support.AnnotationConfigWebApplicationContext;
016    import org.springframework.web.context.support.GenericWebApplicationContext;
017    import org.springframework.web.filter.CharacterEncodingFilter;
018    import org.springframework.web.filter.HiddenHttpMethodFilter;
019    import org.springframework.web.servlet.DispatcherServlet;
020    
021    import com.opensymphony.sitemesh.webapp.SiteMeshFilter;
022    
023    /**
024     * Inspired by http://rockhoppertech.com/blog/spring-mvc-configuration-without-xml/
025     * 
026     * @author LanyonM
027     */
028    public class WebAppInitializer implements WebApplicationInitializer {
029    
030            private static final Logger log = LoggerFactory.getLogger(WebAppInitializer.class);
031    
032            public void onStartup(ServletContext servletContext) throws ServletException {
033                    // Create the 'root' Spring application context
034                    AnnotationConfigWebApplicationContext root = new AnnotationConfigWebApplicationContext();
035                    root.scan("org.lanyonm.grabbag");
036                    root.getEnvironment().setDefaultProfiles("embedded");
037                    
038                    // Manage the lifecycle of the root appcontext
039                    servletContext.addListener(new ContextLoaderListener(root));
040                    servletContext.setInitParameter("defaultHtmlEscape", "true");
041                    
042                    // now the config for the Dispatcher servlet
043                    AnnotationConfigWebApplicationContext mvcContext = new AnnotationConfigWebApplicationContext();
044                    mvcContext.register(WebConfig.class);
045                    
046                    // Filters - http://static.springsource.org/spring/docs/3.2.x/javadoc-api/org/springframework/web/filter/package-frame.html
047                    // Hidden Http Methods
048    //              FilterRegistration.Dynamic fr = servletContext.addFilter("hiddenHttpMethodFilter", new HiddenHttpMethodFilter());
049    //              fr.addMappingForUrlPatterns(null, true, "/*");
050    
051                    // UTF-8 Encoding
052                    FilterRegistration.Dynamic fr = servletContext.addFilter("encodingFilter", new CharacterEncodingFilter());
053                    fr.setInitParameter("encoding", "UTF-8");
054                    fr.setInitParameter("forceEncoding", "true");
055                    fr.addMappingForUrlPatterns(null, true, "/*");
056                    
057                    // SiteMesh
058                    fr = servletContext.addFilter("sitemesh", new SiteMeshFilter());
059                    fr.addMappingForUrlPatterns(null, true, "/*");
060                    
061                    // The main Spring MVC servlet
062                    ServletRegistration.Dynamic dispatcherServlet = servletContext.addServlet(
063                                    "dispatcher", new DispatcherServlet(new GenericWebApplicationContext()));
064                    dispatcherServlet.setLoadOnStartup(1);
065                    Set<String> mappingConflicts = dispatcherServlet.addMapping("/");
066                    if (!mappingConflicts.isEmpty()) {
067                            for (String s : mappingConflicts) {
068                                    log.error("Mapping conflict: " + s);
069                            }
070                            throw new IllegalStateException("'dispatcher' could not be mapped to '/' due " +
071                                            "to an existing mapping. This is a known issue under Tomcat versions " +
072                                            "<= 7.0.14; see https://issues.apache.org/bugzilla/show_bug.cgi?id=51278");
073                    }
074                    
075                    // set various application-wide variables
076                    servletContext.setAttribute("year", Calendar.getInstance().get(Calendar.YEAR));
077            }
078    
079    }
