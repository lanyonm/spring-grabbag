001    package org.lanyonm.grabbag.config;
002    
003    import javax.sql.DataSource;
004    
005    import org.apache.ibatis.session.SqlSessionFactory;
006    import org.mybatis.spring.SqlSessionFactoryBean;
007    import org.mybatis.spring.annotation.MapperScan;
008    
009    import org.springframework.context.annotation.Bean;
010    import org.springframework.context.annotation.Configuration;
011    import org.springframework.jdbc.datasource.DataSourceTransactionManager;
012    //import org.springframework.context.annotation.Profile;
013    import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseBuilder;
014    import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseType;
015    
016    /**
017     * This class is used to configure all the repository connections for the app.
018     * 
019     * @author LanyonM
020     */
021    @Configuration
022    //@Profile({"dev", "test"})
023    @MapperScan("org.lanyonm.grabbag.persistence")
024    public class DataConfig {
025    
026            @Bean
027            public DataSource dataSource() {
028                    return new EmbeddedDatabaseBuilder()
029                                    .setType(EmbeddedDatabaseType.H2)
030                                    .addScript("database/schema.sql")
031                                    .addScript("database/init.sql")
032                                    .build();
033            }
034    
035            @Bean
036            public DataSourceTransactionManager transactionManager() {
037                    return new DataSourceTransactionManager(dataSource());
038            }
039    
040            @Bean
041            public SqlSessionFactory sqlSessionFactory() throws Exception {
042                    SqlSessionFactoryBean sessionFactory = new SqlSessionFactoryBean();
043                    sessionFactory.setDataSource(dataSource());
044                    sessionFactory.setTypeAliasesPackage("org.lanyonm.grabbag.domain");
045                    return sessionFactory.getObject();
046            }
047    
048    }
