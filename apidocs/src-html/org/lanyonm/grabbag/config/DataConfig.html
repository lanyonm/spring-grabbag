001    package org.lanyonm.grabbag.config;
002    
003    import javax.sql.DataSource;
004    
005    import org.apache.ibatis.session.SqlSessionFactory;
006    import org.mybatis.spring.SqlSessionFactoryBean;
007    import org.mybatis.spring.annotation.MapperScan;
008    
009    import org.springframework.context.annotation.Bean;
010    import org.springframework.context.annotation.Configuration;
011    import org.springframework.jdbc.datasource.DataSourceTransactionManager;
012    //import org.springframework.context.annotation.Profile;
013    import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseBuilder;
014    import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseType;
015    
016    @Configuration
017    //@Profile({"dev", "test"})
018    @MapperScan("org.lanyonm.grabbag.persistence")
019    public class DataConfig {
020    
021            @Bean
022            public DataSource dataSource() {
023                    return new EmbeddedDatabaseBuilder()
024                                    .setType(EmbeddedDatabaseType.H2)
025                                    .addScript("database/schema.sql")
026                                    .addScript("database/init.sql")
027                                    .build();
028            }
029    
030            @Bean
031            public DataSourceTransactionManager transactionManager() {
032                    return new DataSourceTransactionManager(dataSource());
033            }
034    
035            @Bean
036            public SqlSessionFactory sqlSessionFactory() throws Exception {
037                    SqlSessionFactoryBean sessionFactory = new SqlSessionFactoryBean();
038                    sessionFactory.setDataSource(dataSource());
039                    sessionFactory.setTypeAliasesPackage("org.lanyonm.grabbag.domain");
040                    return sessionFactory.getObject();
041            }
042    
043    }
