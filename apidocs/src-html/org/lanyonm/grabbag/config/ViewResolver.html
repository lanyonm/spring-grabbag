001    package org.lanyonm.grabbag.config;
002    
003    import org.springframework.stereotype.Component;
004    import org.springframework.web.servlet.view.AbstractUrlBasedView;
005    import org.springframework.web.servlet.view.InternalResourceViewResolver;
006    
007    /**
008     * Dynamic view resolver.
009     * 
010     * @author lanyonm
011     *
012     */
013    @Component
014    public class ViewResolver extends InternalResourceViewResolver {
015            
016            /**
017             * Use '/WEB-INF/views/' for the prefix and '.jsp' for the suffix.
018             */
019            public ViewResolver() {
020                    super();
021                    setPrefix("/WEB-INF/views/");
022                    setSuffix(".jsp");
023            }
024    
025            @Override
026            protected AbstractUrlBasedView buildView(String viewName) throws Exception {
027                    if (viewName.isEmpty() || viewName.endsWith("/")) {
028                            viewName += "index";
029                    }
030                    return super.buildView(viewName);
031            }
032    
033    }
