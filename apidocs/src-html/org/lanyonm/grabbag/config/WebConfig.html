001    package org.lanyonm.grabbag.config;
002    
003    import java.util.Locale;
004    
005    import org.slf4j.Logger;
006    import org.slf4j.LoggerFactory;
007    import org.springframework.context.MessageSource;
008    import org.springframework.context.annotation.Bean;
009    import org.springframework.context.annotation.Configuration;
010    import org.springframework.context.support.ReloadableResourceBundleMessageSource;
011    import org.springframework.web.servlet.LocaleResolver;
012    import org.springframework.web.servlet.config.annotation.EnableWebMvc;
013    import org.springframework.web.servlet.config.annotation.InterceptorRegistry;
014    import org.springframework.web.servlet.config.annotation.WebMvcConfigurerAdapter;
015    import org.springframework.web.servlet.i18n.LocaleChangeInterceptor;
016    import org.springframework.web.servlet.i18n.SessionLocaleResolver;
017    
018    /**
019     * A config class for all the web components
020     * 
021     * @author LanyonM
022     */
023    @Configuration
024    @EnableWebMvc
025    public class WebConfig extends WebMvcConfigurerAdapter {
026    
027            private static final String MESSAGE_SOURCE = "classpath:i18n/messages";
028            private static final Logger log = LoggerFactory.getLogger(WebConfig.class);
029    
030            @Bean
031            public MessageSource messageSource() {
032                    log.debug("setting up message source");
033                    ReloadableResourceBundleMessageSource messageSource = new ReloadableResourceBundleMessageSource();
034                    messageSource.setBasename(MESSAGE_SOURCE);
035                    messageSource.setCacheSeconds(5);
036                    messageSource.setDefaultEncoding("UTF-8");
037                    return messageSource;
038            }
039    
040            @Bean
041            public LocaleResolver localeResolver() {
042                    SessionLocaleResolver lr = new SessionLocaleResolver();
043    //              lr.setDefaultLocale(Locale.ENGLISH);
044                    return lr;
045            }
046    
047            @Override
048            public void addInterceptors(InterceptorRegistry registry) {
049    //              registry.addInterceptor(new LocaleChangeInterceptor());
050            }
051    }
