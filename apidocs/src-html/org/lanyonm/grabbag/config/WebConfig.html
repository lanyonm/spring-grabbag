001    package org.lanyonm.grabbag.config;
002    
003    import java.util.Locale;
004    
005    import org.slf4j.Logger;
006    import org.slf4j.LoggerFactory;
007    import org.springframework.context.MessageSource;
008    import org.springframework.context.annotation.Bean;
009    import org.springframework.context.annotation.Configuration;
010    import org.springframework.context.support.ReloadableResourceBundleMessageSource;
011    import org.springframework.web.servlet.LocaleResolver;
012    import org.springframework.web.servlet.config.annotation.EnableWebMvc;
013    import org.springframework.web.servlet.config.annotation.InterceptorRegistry;
014    import org.springframework.web.servlet.config.annotation.ResourceHandlerRegistry;
015    import org.springframework.web.servlet.config.annotation.WebMvcConfigurerAdapter;
016    import org.springframework.web.servlet.i18n.LocaleChangeInterceptor;
017    import org.springframework.web.servlet.i18n.SessionLocaleResolver;
018    
019    /**
020     * A config class for all the web components
021     *
022     * @author LanyonM
023     */
024    @Configuration
025    @EnableWebMvc
026    public class WebConfig extends WebMvcConfigurerAdapter {
027    
028            private static final String MESSAGE_SOURCE = "classpath:i18n/messages";
029            private static final Logger log = LoggerFactory.getLogger(WebConfig.class);
030    
031            @Override
032            public void addResourceHandlers(ResourceHandlerRegistry registry) {
033                    registry.addResourceHandler("/static/**").addResourceLocations("/static/");
034            }
035    
036            @Bean
037            public MessageSource messageSource() {
038                    log.debug("setting up message source");
039                    ReloadableResourceBundleMessageSource messageSource = new ReloadableResourceBundleMessageSource();
040                    messageSource.setBasename(MESSAGE_SOURCE);
041                    messageSource.setCacheSeconds(5);
042                    messageSource.setDefaultEncoding("UTF-8");
043                    return messageSource;
044            }
045    
046            @Bean
047            public LocaleResolver localeResolver() {
048                    SessionLocaleResolver lr = new SessionLocaleResolver();
049                    // lr.setDefaultLocale(Locale.ENGLISH);
050                    return lr;
051            }
052    
053            @Override
054            public void addInterceptors(InterceptorRegistry registry) {
055    //              registry.addInterceptor(new LocaleChangeInterceptor());
056            }
057    }
