001    package org.lanyonm.grabbag.config;
002    
003    import org.springframework.context.annotation.Configuration;
004    
005    import com.codahale.metrics.JmxReporter;
006    import com.codahale.metrics.MetricRegistry;
007    import com.codahale.metrics.SharedMetricRegistries;
008    import com.ryantenney.metrics.spring.config.annotation.EnableMetrics;
009    import com.ryantenney.metrics.spring.config.annotation.MetricsConfigurerAdapter;
010    
011    /**
012     * This class is used to configure any and all metrics collection from within the app.
013     * 
014     * @author LanyonM
015     */
016    @Configuration
017    @EnableMetrics
018    public class MetricsConfig extends MetricsConfigurerAdapter {
019    
020            @Override
021            public MetricRegistry getMetricRegistry() {
022                    return SharedMetricRegistries.getOrCreate("springMetrics");
023            }
024    
025            @Override
026            public void configureReporters(MetricRegistry registry) {
027                    // final ConsoleReporter reporter = ConsoleReporter.forRegistry(registry)
028                    //              .convertRatesTo(TimeUnit.SECONDS)
029                    //              .convertDurationsTo(TimeUnit.MILLISECONDS)
030                    //              .build();
031                    // reporter.start(1, TimeUnit.MINUTES);
032                    final JmxReporter reporter = JmxReporter.forRegistry(registry).build();
033                    reporter.start();
034            }
035    }
