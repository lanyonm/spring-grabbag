001    package org.lanyonm.grabbag.service;
002    
003    import java.util.List;
004    
005    import org.lanyonm.grabbag.domain.Ingredient;
006    import org.lanyonm.grabbag.domain.Recipe;
007    import org.lanyonm.grabbag.persistence.IngredientMapper;
008    import org.lanyonm.grabbag.persistence.RecipeMapper;
009    import org.springframework.beans.factory.annotation.Autowired;
010    import org.springframework.stereotype.Service;
011    
012    @Service("cookbookService")
013    public class CookbookServiceImpl implements CookbookService {
014    
015            @Autowired
016            private RecipeMapper recipeMapper;
017            @Autowired
018            private IngredientMapper ingredientMapper;
019    
020            public List<Recipe> getAllRecipes() {
021                    return recipeMapper.getAllRecipes();
022            }
023    
024            public Recipe getRecipe(int id) {
025                    return recipeMapper.getRecipe(id);
026            }
027    
028            public boolean saveRecipe(Recipe recipe) {
029                    int ret = 0;
030                    if (recipe.getId() > 0) {
031                            ret = recipeMapper.updateRecipe(recipe);
032                    } else {
033                            ret = recipeMapper.insertRecipe(recipe);
034                    }
035                    return ret == 1;
036            }
037    
038            public boolean deleteRecipe(Recipe recipe) {
039                    return recipeMapper.deleteRecipe(recipe) > 0;
040            }
041    
042            public List<Ingredient> getAllIngredients() {
043                    return ingredientMapper.getAllIngredients();
044            }
045    
046            public Ingredient getIngredient(int id) {
047                    return ingredientMapper.getIngredient(id);
048            }
049    
050            public boolean saveIngredient(Ingredient ingredient) {
051                    int ret = 0;
052                    if (ingredient.getId() > 0) {
053                            ret = ingredientMapper.updateIngredient(ingredient);
054                    } else {
055                            ret = ingredientMapper.insertIngredient(ingredient);
056                    }
057                    return ret == 1;
058            }
059    
060            public boolean deleteIngredient(Ingredient ingredient) {
061                    if (!getRecipesWithIngredient(ingredient).isEmpty()) {
062                            return false;
063                    } else {
064                            return ingredientMapper.deleteIngredient(ingredient) > 0;
065                    }
066            }
067    
068            public List<Recipe> getRecipesWithIngredient(Ingredient ingredient) {
069                    return recipeMapper.getRecipeWithIngredient(ingredient);
070            }
071    
072    }
