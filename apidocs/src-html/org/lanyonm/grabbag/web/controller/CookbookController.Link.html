066            @RequestMapping(value = "/recipe/{id}/edit", method = RequestMethod.GET)
067            public String recipeAddEditShow(@PathVariable("id") final int id, Model model) {
068                    Recipe recipe = cookbookService.getRecipe(id);
069                    if (recipe == null && id > 0) {
070                            model.addAttribute("error", "Something nefarious was attempted.");
071                            return "redirect:/cookbook/";
072                    } else if (id == 0) {
073                            recipe = new Recipe();
074                    }
075                    model.addAttribute("recipe", recipe);
076                    model.addAttribute("allIngredients", cookbookService.getAllIngredients());
077                    return "cookbook/recipeAddEdit";
078            }

080            @RequestMapping(value = "/recipe/{id}/edit", method = RequestMethod.POST)
081            public String recipeAddEditSave(@PathVariable("id") final int id, Model model, @ModelAttribute("recipe") RecipeForm recipeForm, BindingResult result) {
082                    Recipe recipe = new Recipe();
083                    if (id == 0) {
084                            recipe.setName(recipeForm.getName());
085                            recipe.setDescription(recipeForm.getDescription());
086                            cookbookService.saveRecipe(recipe);
087                    } else {
088                            recipe = cookbookService.getRecipe(id);
089                            if (recipe != null) {
090                                    recipe.setName(recipeForm.getName());
091                                    recipe.setDescription(recipeForm.getDescription());
092                                    cookbookService.saveRecipe(recipe);
093                            } else {
094                                    model.addAttribute("error", "Something nefarious was attempted.");
095                            }
096                    }
097                    return "redirect:/cookbook/";
098            }

119                    Ingredient ingredient = cookbookService.getIngredient(id);

122                            model.addAttribute("error", "Something nefarious was attempted.");

132            @RequestMapping(value = "/ingredient/{id}/edit", method = RequestMethod.POST)
133            public String ingredientAddEditSave(@PathVariable("id") final int id, Model model, @ModelAttribute("ingredient") IngredientForm ingredientForm, BindingResult result) {
134                    model.addAttribute("ingredient", ingredientForm);
135                    validator.validate(ingredientForm, result);
136                    if (result.hasErrors()) {
137                            model.addAttribute("isNew", id == 0);
138                            return "cookbook/ingredientAddEdit";
139                    }
140    
141                    Ingredient ingredient = new Ingredient();
142                    if (id == 0) {
143                            ingredient.setName(ingredientForm.getName());
144                            ingredient.setDescription(ingredientForm.getDescription());
145                            cookbookService.saveIngredient(ingredient);
146                    } else {
147                            ingredient = cookbookService.getIngredient(id);
148                            if (ingredient != null) {
149                                    ingredient.setName(ingredientForm.getName());
150                                    ingredient.setDescription(ingredientForm.getDescription());
151                                    cookbookService.saveIngredient(ingredient);
152                            } else {
153                                    model.addAttribute("error", "Something nefarious was attempted.");
154                            }
155                    }
156                    return "redirect:/cookbook/ingredients";
157            }

159            @RequestMapping(value = "/ingredient/{id}/delete", method = RequestMethod.GET)
160            public String ingredientDelete(@PathVariable("id") final int id, Model model) {
161                    Ingredient ingredient = cookbookService.getIngredient(id);
162                    if (!cookbookService.getRecipesWithIngredient(ingredient).isEmpty()) {
163                            List<Recipe> recipes = cookbookService.getRecipesWithIngredient(ingredient);
164                            StringBuilder builder = new StringBuilder("Couldn't delete the ingredient. ");
165                            for (Recipe recipe : recipes) {
166                                    builder.append("It is used in " + recipe.getName() + ". ");
167                            }
168                            model.addAttribute("error", builder.toString());
169                    } else {
170                            if (cookbookService.deleteIngredient(ingredient)) {
171                                    model.addAttribute("message", "You successfully deleted " + ingredient.getName() + "!");
172                            } else {
173                                    model.addAttribute("error", "There was an error deleteing the ingredient.");
174                            }
175                    }
176                    return "redirect:/cookbook/ingredients";
177            }

179            /**
180             * This method supplies json for D3
181             * 
182             * @param model
183             * @return
184             */
185            @RequestMapping(value = "/graph.json", method = RequestMethod.GET)
186            public String graphData(Model model, HttpServletResponse response) {
187                    List<Recipe> recipes = cookbookService.getAllRecipes();
188                    List<Ingredient> ingredients = cookbookService.getAllIngredients();
189                    List<Node> nodes = new ArrayList<Node>();
190                    for (Recipe recipe : recipes) {
191                            nodes.add(new Node(recipe.getName(), 1));
192                    }
193                    for (Ingredient ingredient : ingredients) {
194                            nodes.add(new Node(ingredient.getName(), 2));
195                    }
196                    model.addAttribute("nodes", nodes);
197                    List<Link> links = new ArrayList<Link>();
198                    
199                    int rCount = 0;
200                    for (Recipe recipe : recipes) {
201                            for (Ingredient ri : recipe.getIngredients()) {
202                                    Link link = new Link(rCount, ingredients.indexOf(ri) + recipes.size(), 1);
203                                    log.debug(recipe.getName() + " and " + ri.getName() + " is: " + link);
204                                    links.add(link);
205                            }
206                            rCount++;
207                    }
208                    model.addAttribute("links", links);
209                    response.setContentType(MediaType.APPLICATION_JSON_VALUE);
210                    return "cookbook/graph";
211            }

220            public class Node {
221                    private String name;
222                    private int group;
223                    public Node(String name, int group) {
224                            this.name = name;
225                            this.group = group;
226                    }
227                    public String getName() {
228                            return this.name;
229                    }
230                    public int getGroup() {
231                            return this.group;
232                    }
233            }
234    
235            public class Link {
236                    private int source;
237                    private int target;
238                    private int value;
239                    public Link(int source, int target, int value) {
240                            this.source = source;
241                            this.target = target;
242                            this.value = value;
243                    }
244                    public int getSource() {
245                            return this.source;
246                    }
247                    public int getTarget() {
248                            return this.target;
249                    }
250                    public int getValue() {
251                            return this.value;
252                    }
253                    public String toString() {
254                            StringBuilder sb = new StringBuilder("Link = [");
255                            sb.append("source=").append(source)
256                                    .append(", target=").append(target)
257                                    .append(", value=").append(value).append("]");
258                            return sb.toString();
259                    }
260            }
