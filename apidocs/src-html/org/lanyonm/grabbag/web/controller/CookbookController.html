001    package org.lanyonm.grabbag.web.controller;
002    
003    import java.util.List;
004    
005    import org.lanyonm.grabbag.web.form.IngredientForm;
006    import org.lanyonm.grabbag.web.form.RecipeForm;
007    import org.lanyonm.grabbag.domain.Ingredient;
008    import org.lanyonm.grabbag.domain.Recipe;
009    import org.lanyonm.grabbag.service.CookbookService;
010    
011    import org.slf4j.Logger;
012    import org.slf4j.LoggerFactory;
013    import org.springframework.beans.factory.annotation.Autowired;
014    import org.springframework.stereotype.Controller;
015    import org.springframework.ui.Model;
016    import org.springframework.validation.BindingResult;
017    import org.springframework.validation.Validator;
018    import org.springframework.web.bind.annotation.ModelAttribute;
019    import org.springframework.web.bind.annotation.PathVariable;
020    import org.springframework.web.bind.annotation.RequestMapping;
021    import org.springframework.web.bind.annotation.RequestMethod;
022    
023    @Controller
024    @RequestMapping("/cookbook")
025    public class CookbookController {
026    
027            @Autowired
028            private CookbookService cookbookService;
029            @Autowired
030            private Validator validator;
031    
032            private static final Logger log = LoggerFactory.getLogger(CookbookController.class);
033    
034            @RequestMapping("/")
035            public String index(Model model) {
036                    log.debug("Cookbook index mapping");
037                    model.addAttribute("recipes", cookbookService.getAllRecipes());
038                    return "cookbook/index";
039            }
040    
041            @RequestMapping(value = "/recipe/{id}", method = RequestMethod.GET)
042            public String recipe(@PathVariable("id") final int id, Model model) {
043                    Recipe recipe = cookbookService.getRecipe(id);
044                    if (recipe == null) {
045                            model.addAttribute("error", "Something nefarious was attempted.");
046                            return "redirect:/cookbook/";
047                    }
048                    model.addAttribute("recipe", recipe);
049                    return "cookbook/recipe";
050            }
051    
052            @RequestMapping(value = "/recipe/{id}/edit", method = RequestMethod.GET)
053            public String recipeAddEditShow(@PathVariable("id") final int id, Model model) {
054                    Recipe recipe = cookbookService.getRecipe(id);
055                    if (recipe == null && id > 0) {
056                            model.addAttribute("error", "Something nefarious was attempted.");
057                            return "redirect:/cookbook/";
058                    } else if (id == 0) {
059                            recipe = new Recipe();
060                    }
061                    model.addAttribute("recipe", recipe);
062                    model.addAttribute("allIngredients", cookbookService.getAllIngredients());
063                    return "cookbook/recipeAddEdit";
064            }
065    
066            @RequestMapping(value = "/recipe/{id}/edit", method = RequestMethod.POST)
067            public String recipeAddEditSave(@PathVariable("id") final int id, Model model, @ModelAttribute("recipe") RecipeForm recipeForm, BindingResult result) {
068                    Recipe recipe = new Recipe();
069                    if (id == 0) {
070                            recipe.setName(recipeForm.getName());
071                            recipe.setDescription(recipeForm.getDescription());
072                            cookbookService.saveRecipe(recipe);
073                    } else {
074                            recipe = cookbookService.getRecipe(id);
075                            if (recipe != null) {
076                                    recipe.setName(recipeForm.getName());
077                                    recipe.setDescription(recipeForm.getDescription());
078                                    cookbookService.saveRecipe(recipe);
079                            } else {
080                                    model.addAttribute("error", "Something nefarious was attempted.");
081                            }
082                    }
083                    return "redirect:/cookbook/";
084            }
085    
086            @RequestMapping(value = "/recipe/{id}/delete", method = RequestMethod.GET)
087            public String deleteRecipe(@PathVariable("id") final int id, Model model) {
088                    Recipe recipe = cookbookService.getRecipe(id);
089                    if (cookbookService.deleteRecipe(recipe)) {
090                            model.addAttribute("message", "You successfully deleted " + recipe.getName() + "!");
091                    } else {
092                            model.addAttribute("error", "There was an error deleting the recipe.");
093                    }
094                    return "redirect:/cookbook/";
095            }
096    
097            @RequestMapping("/ingredients")
098            public String ingredients(Model model) {
099                    model.addAttribute("ingredients", cookbookService.getAllIngredients());
100                    return "cookbook/ingredients";
101            }
102    
103            @RequestMapping(value = "/ingredient/{id}/edit", method = RequestMethod.GET)
104            public String ingredientAddEditShow(@PathVariable("id") final int id, Model model) {
105                    Ingredient ingredient = cookbookService.getIngredient(id);
106                    log.debug("when searching for ingredient " + id + ", we found " + ingredient);
107                    if (ingredient == null && id > 0) {
108                            model.addAttribute("error", "Something nefarious was attempted.");
109                            return "redirect:/cookbook/ingredients";
110                    } else if (ingredient == null && id == 0) {
111                            ingredient = new Ingredient();
112                    }
113                    model.addAttribute("ingredient", ingredient);
114                    model.addAttribute("isNew", id == 0);
115                    return "cookbook/ingredientAddEdit";
116            }
117    
118            @RequestMapping(value = "/ingredient/{id}/edit", method = RequestMethod.POST)
119            public String ingredientAddEditSave(@PathVariable("id") final int id, Model model, @ModelAttribute("ingredient") IngredientForm ingredientForm, BindingResult result) {
120                    model.addAttribute("ingredient", ingredientForm);
121                    validator.validate(ingredientForm, result);
122                    if (result.hasErrors()) {
123                            model.addAttribute("isNew", id == 0);
124                            return "cookbook/ingredientAddEdit";
125                    }
126    
127                    Ingredient ingredient = new Ingredient();
128                    if (id == 0) {
129                            ingredient.setName(ingredientForm.getName());
130                            ingredient.setDescription(ingredientForm.getDescription());
131                            cookbookService.saveIngredient(ingredient);
132                    } else {
133                            ingredient = cookbookService.getIngredient(id);
134                            if (ingredient != null) {
135                                    ingredient.setName(ingredientForm.getName());
136                                    ingredient.setDescription(ingredientForm.getDescription());
137                                    cookbookService.saveIngredient(ingredient);
138                            } else {
139                                    model.addAttribute("error", "Something nefarious was attempted.");
140                            }
141                    }
142                    return "redirect:/cookbook/ingredients";
143            }
144    
145            @RequestMapping(value = "/ingredient/{id}/delete", method = RequestMethod.GET)
146            public String ingredientDelete(@PathVariable("id") final int id, Model model) {
147                    Ingredient ingredient = cookbookService.getIngredient(id);
148                    if (!cookbookService.getRecipesWithIngredient(ingredient).isEmpty()) {
149                            List<Recipe> recipes = cookbookService.getRecipesWithIngredient(ingredient);
150                            StringBuilder builder = new StringBuilder("Couldn't delete the ingredient. ");
151                            for (Recipe recipe : recipes) {
152                                    builder.append("It is used in " + recipe.getName() + ". ");
153                            }
154                            model.addAttribute("error", builder.toString());
155                    } else {
156                            if (cookbookService.deleteIngredient(ingredient)) {
157                                    model.addAttribute("message", "You successfully deleted " + ingredient.getName() + "!");
158                            } else {
159                                    model.addAttribute("error", "There was an error deleteing the ingredient.");
160                            }
161                    }
162                    return "redirect:/cookbook/ingredients";
163            }
164    }
