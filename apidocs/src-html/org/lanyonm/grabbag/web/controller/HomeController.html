001    package org.lanyonm.grabbag.web.controller;
002    
003    import org.lanyonm.grabbag.dao.UsersDao;
004    import org.lanyonm.grabbag.persistence.IngredientMapper;
005    import org.slf4j.Logger;
006    import org.slf4j.LoggerFactory;
007    import org.springframework.beans.factory.annotation.Autowired;
008    import org.springframework.stereotype.Controller;
009    import org.springframework.ui.Model;
010    import org.springframework.web.bind.annotation.RequestMapping;
011    import org.springframework.web.bind.annotation.RequestMethod;
012    
013    @Controller
014    public class HomeController {
015    
016            @Autowired
017            private UsersDao usersDao;
018    
019            @Autowired
020            private IngredientMapper ingredientMapper;
021            
022            private static final Logger log = LoggerFactory.getLogger(HomeController.class);
023    
024            /**
025             * A basic homepage mapping.  This controller action is passing a string
026             * to the view.
027             * 
028             * @param model
029             */
030            @RequestMapping(value = "/", method = RequestMethod.GET)
031            public void index(Model model) {
032                    log.debug("In the HomeController::index");
033                    model.addAttribute("sample", "Hello World!");
034            }
035    
036            @RequestMapping("/defined-path")
037            public String definedPath(Model model) {
038                    log.debug("hit the defined path");
039                    String viewPath = "testView";
040                    model.addAttribute("sample", "Defined Path!!!");
041                    return viewPath;
042            }
043            
044            @RequestMapping(value = "/jdbc", method = RequestMethod.GET)
045            public void jdbc(Model model) {
046                    model.addAttribute("users", usersDao.getUsers());
047            }
048    
049            @RequestMapping(value = "/mybatis", method = RequestMethod.GET)
050            public void mybatis(Model model) {
051                    model.addAttribute("ingredients", ingredientMapper.getAllIngredients());
052            }
053    }
