001    package org.lanyonm.grabbag.dao.jdbc;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    import java.util.Map;
006    
007    import javax.sql.DataSource;
008    
009    import org.lanyonm.grabbag.dao.UsersDao;
010    import org.lanyonm.grabbag.domain.User;
011    import org.springframework.beans.factory.annotation.Autowired;
012    import org.springframework.jdbc.core.JdbcTemplate;
013    import org.springframework.stereotype.Repository;
014    
015    @Repository("usersDao")
016    public class UsersDaoImpl implements UsersDao {
017    
018            @Autowired
019            private DataSource dataSource;
020    
021            /*
022             * (non-Javadoc)
023             * @see org.lanyonm.grabbag.dao.UsersDao#getUsers()
024             */
025            public List<User> getUsers() {
026                    JdbcTemplate template = new JdbcTemplate(dataSource);
027                    List<Map<String, Object>> ret = template.queryForList("select * from user;");
028                    List<User> users = new ArrayList<User>();
029                    for (Map<String, Object> row : ret) {
030                            users.add(new User((Long) row.get("ID"), row.get("FIRST_NAME").toString(), row.get("LAST_NAME").toString(), row.get("EMAIL").toString()));
031                    }
032                    return users;
033            }
034    }
