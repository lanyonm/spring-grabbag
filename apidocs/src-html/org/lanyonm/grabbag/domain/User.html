001    package org.lanyonm.grabbag.domain;
002    
003    import java.io.Serializable;
004    
005    /**
006     * 
007     * @author mlanyon
008     */
009    public class User implements Serializable {
010    
011            private long id;
012            private String firstName;
013            private String lastName;
014            private String email;
015            private static final long serialVersionUID = 1L;
016    
017            public User() { }
018    
019            public User(long id, String firstName, String lastName, String email) {
020                    this.id = id;
021                    this.firstName = firstName;
022                    this.lastName = lastName;
023                    this.email = email;
024            }
025    
026            public long getId() {
027                    return id;
028            }
029    
030            public String getFirstName() {
031                    return firstName;
032            }
033    
034            public String getLastName() {
035                    return lastName;
036            }
037    
038            public String getEmail() {
039                    return email;
040            }
041    
042            @Override
043            public String toString() {
044                    StringBuilder sb = new StringBuilder();
045                    sb.append(getFirstName()).append(" ").append(getLastName());
046                    sb.append(" <").append(getEmail()).append(">");
047                    return sb.toString();
048            }
049    }
